const inputGlb = document.getElementById('inputGlb'); const inputUsdz = document.getElementById('inputUsdz'); const boxGlb = document.getElementById('boxGlb'); const boxUsdz = document.getElementById('boxUsdz'); const lblGlb = document.getElementById('lblGlb'); const lblUsdz = document.getElementById('lblUsdz'); const uploadBtn = document.getElementById('uploadBtn'); const status = document.getElementById('status'); const shareSection = document.getElementById('shareSection'); const shareUrl = document.getElementById('shareUrl'); const timerBadge = document.getElementById('timerBadge'); const countdown = document.getElementById('countdown'); const viewerContainer = document.getElementById('viewerContainer'); const MAX_SIZE = 10 * 1024 * 1024; // 10MB let localGlbUrl = ""; let localUsdzUrl = ""; let timerInterval = null; // Menyimpan referensi interval agar tidak double // INSTAN PREVIEW SAAT FILE GLB DIPILIH inputGlb.addEventListener('change', () => { if (inputGlb.files[0]?.size > MAX_SIZE) { alert("File GLB lebih dari 10MB!"); inputGlb.value=""; return; } if (inputGlb.files[0]) { boxGlb.classList.add('has-file'); lblGlb.innerText = inputGlb.files[0].name; if (localGlbUrl) URL.revokeObjectURL(localGlbUrl); localGlbUrl = URL.createObjectURL(inputGlb.files[0]); initEngineViewer(localGlbUrl, localUsdzUrl); status.innerHTML = "Local Preview Aktif. Siap di-view/di-AR!"; } checkInputs(); }); // INSTAN PREVIEW UNTUK USDZ inputUsdz.addEventListener('change', () => { if (inputUsdz.files[0]?.size > MAX_SIZE) { alert("File USDZ lebih dari 10MB!"); inputUsdz.value=""; return; } if (inputUsdz.files[0]) { boxUsdz.classList.add('has-file'); lblUsdz.innerText = inputUsdz.files[0].name; if (localUsdzUrl) URL.revokeObjectURL(localUsdzUrl); localUsdzUrl = URL.createObjectURL(inputUsdz.files[0]); if (localGlbUrl) { initEngineViewer(localGlbUrl, localUsdzUrl); } } checkInputs(); }); function checkInputs() { uploadBtn.disabled = !inputGlb.files[0]; } window.addEventListener('DOMContentLoaded', () => { const params = new URLSearchParams(window.location.search); const modelParam = params.get('glb'); const iosParam = params.get('usdz'); if (modelParam) { initEngineViewer(modelParam, iosParam); startDestructionTimer(); } }); // TOMBOL PUBLISH SHARE uploadBtn.addEventListener('click', async () => { uploadBtn.disabled = true; status.innerHTML = "Mengirim Manifest Permanen ke Cloud Vercel..."; try { // 1. Upload GLB const resGlb = await fetch(`/api/upload?filename=${encodeURIComponent(inputGlb.files[0].name)}`, { method: 'POST', body: inputGlb.files[0] }); if (!resGlb.ok) throw new Error("Gagal mengunggah file GLB"); const dataGlb = await resGlb.json(); // 2. Upload USDZ jika ada let cloudUsdzUrl = ""; if (inputUsdz.files[0]) { status.innerHTML = "Menyinkronkan iOS Engine ke Cloud..."; const resUsdz = await fetch(`/api/upload?filename=${encodeURIComponent(inputUsdz.files[0].name)}`, { method: 'POST', body: inputUsdz.files[0] }); if (!resUsdz.ok) throw new Error("Gagal mengunggah file USDZ"); const dataUsdz = await resUsdz.json(); cloudUsdzUrl = dataUsdz.url; } status.innerHTML = "Published! File tersimpan di Cloud sampai tengah malam."; let shareLink = `${window.location.origin}${window.location.pathname}?glb=${encodeURIComponent(dataGlb.url)}`; if(cloudUsdzUrl) shareLink += `&usdz=${encodeURIComponent(cloudUsdzUrl)}`; shareUrl.innerText = shareLink; shareSection.style.display = 'block'; // Bersihkan Object URL lokal karena sudah digantikan link cloud (Hemat Memori) if (localGlbUrl) { URL.revokeObjectURL(localGlbUrl); localGlbUrl = ""; } if (localUsdzUrl) { URL.revokeObjectURL(localUsdzUrl); localUsdzUrl = ""; } initEngineViewer(dataGlb.url, cloudUsdzUrl); startDestructionTimer(); } catch (err) { status.innerHTML = `Error Cloud: ${err.message}`; uploadBtn.disabled = false; } }); function initEngineViewer(glbUrl, usdzUrl) { viewerContainer.style.display = 'block'; // Hapus viewer lama jika ada agar tidak tumpang tindih const oldViewer = viewerContainer.querySelector('model-viewer'); if (oldViewer) oldViewer.remove(); const viewer = document.createElement('model-viewer'); viewer.setAttribute('src', glbUrl); viewer.setAttribute('ar', ''); viewer.setAttribute('ar-modes', 'webxr scene-viewer quick-look'); viewer.setAttribute('camera-controls', ''); viewer.setAttribute('auto-rotate', ''); viewer.setAttribute('shadow-intensity', '1'); // Mengatasi bug animasi tidak berputar otomatis saat ganti model viewer.addEventListener('load', () => { setTimeout(() => { const anims = viewer.availableAnimations; if (anims && anims.length > 0) { viewer.setAttribute('animation-name', anims[0]); viewer.play(); } }, 100); // Jeda micro-tasking aman untuk memastikan data animasi ter-load }); if (usdzUrl && usdzUrl !== "null" && usdzUrl !== "") { viewer.setAttribute('ios-src', usdzUrl); } // Masukkan di awal (prepend) agar tidak merusak posisi tombol fullscreen viewerContainer.prepend(viewer); } function startDestructionTimer() { if (timerInterval) clearInterval(timerInterval); // Reset jika ada timer berjalan sebelumnya timerBadge.style.display = 'block'; const updateCountdown = () => { const now = new Date(); const tomorrow = new Date(); tomorrow.setHours(24, 0, 0, 0); const diff = tomorrow - now; if (diff <= 0) { countdown.innerText = "00:00:00"; clearInterval(timerInterval); location.reload(); // Refresh halaman saat waktu habis return; } const hours = String(Math.floor((diff / (1000 * 60 * 60)) % 24)).padStart(2, '0'); const minutes = String(Math.floor((diff / (1000 * 60)) % 60)).padStart(2, '0'); const seconds = String(Math.floor((diff / 1000) % 60)).padStart(2, '0'); countdown.innerText = `${hours}:${minutes}:${seconds}`; }; updateCountdown(); // Jalankan sekali langsung tanpa nunggu 1 detik pertama timerInterval = setInterval(updateCountdown, 1000); } function toggleViewFullscreen() { if (!document.fullscreenElement) { viewerContainer.requestFullscreen().catch(err => { alert(`Gagal Fullscreen: ${err.message}`); }); } else { document.exitFullscreen(); } } function copyUrl() { navigator.clipboard.writeText(shareUrl.innerText).then(() => { alert('Link manifest AR berhasil disalin!'); }).catch(() => { alert('Gagal menyalin link secara otomatis.'); }); }
